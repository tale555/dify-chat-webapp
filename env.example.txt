# Dify API設定
# このファイルを .env にコピーして、実際の値を設定してください
# 
# Windowsの場合: copy env.example.txt .env
# Mac/Linuxの場合: cp env.example.txt .env

# Dify API Key（必須）
# Dify > Settings > API Keys で作成
VITE_DIFY_API_KEY=your_dify_api_key_here

# Dify App ID（必須）
# Dify > アプリ > 設定 > API で確認
VITE_DIFY_APP_ID=your_dify_app_id_here

# Dify API Base URL（オプション）
# デフォルト: https://api.dify.ai/v1
# カスタムドメインを使用する場合は変更
VITE_DIFY_API_BASE_URL=https://api.dify.ai/v1
